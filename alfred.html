the canonical microframework demo, in five lines of trivial code in batman.js

empty index.html

    <!DOCTYPE html>
    <html>
    <head>
        <title>Alfred: Batman's Todos</title>

        <link href='http://fonts.googleapis.com/css?family=Lobster' rel='stylesheet' type='text/css'>
        <link rel="stylesheet" href="resources/alfred.css" type="text/css">

        <script type="text/javascript" src="lib/coffee-script.js"></script>
        <script type="text/javascript" src="lib/batman.js"></script>
    </head>

    <body>
        <div id="container">
            <h1>Alfred</h1>

            <form>
              <input class="new-item" placeholder="add a todo item" />
            </form>

            <ul id="items">
                <li>
                    <input type="checkbox" />
                    <label></label>
                    <a>delete</a>
                </li>
            </ul>
        </div>

        <script type="text/coffeescript">
        class Alfred extends Batman.App

        Alfred.run()
        </script>
    </body>
    </html>

possibly from a designer


define the todo model:

    class Alfred.Todo extends Batman.Model
      body: ''
      isDone: false

    # create a default todo
    if not Todo.all().length
      todo = new Todo body: 'joker escaped from arkham'
      todo.save()

the schema definition is totally optional and can be inferred


show all the todos:

    <ul id="items">
        <li data-foreach-todo="Todo.all">
            <input type="checkbox" />
            <label></label>
            <a>delete</a>
        </li>
    </ul>

forEach todo in Todo.all, create a new <li> that is a copy of this prototype.


show the todo data in the li:

    <ul id="items">
        <li data-foreach-todo="Todo.all">
            <input type="checkbox" data-bind="todo.isDone" />
            <label data-bind="todo.body" data-class-done="todo.isDone"></label>
            <a data-event-click="todo.destroy">delete</a>
        </li>
    </ul>

if todo.isDone is truthy, the class `done` will be added, otherwise it will be removed.


use mixins to easily add functionality:

    <ul id="items">
        <li data-foreach-todo="Todo.all" data-mixin="animation">
            <input type="checkbox" data-bind="todo.isDone" />
            <label data-bind="todo.body" data-class-done="todo.isDone" data-mixin="editable"></label>
            <a data-event-click="todo.destroy">delete</a>
        </li>
    </ul>

whenever a new todo is created, it will slide into view. when you delete one, it will slide out.


easily create bound forms:

    <form data-formfor-todo="Todo.create" data-event-submit="todo.save">
      <input class="new-item" placeholder="add a todo item" data-bind="todo.body" />
    </form>

data-event uses addEventListener and allows you access to your bound context.


persist todos locally:

    class Alfred.Todo extends Batman.Model
      @persist Batman.LocalStorage

also supports Batman.RestStorage, Batman (for the push server)


done!

    <!DOCTYPE html>
    <html>
    <head>
        <title>Alfred: Batman's Todos</title>

        <link href='http://fonts.googleapis.com/css?family=Lobster' rel='stylesheet' type='text/css'>
        <link rel="stylesheet" href="resources/alfred.css" type="text/css">

        <script type="text/javascript" src="lib/coffee-script.js"></script>
        <script type="text/javascript" src="lib/batman.js"></script>
    </head>

    <body>
        <div id="container">
            <h1>Alfred</h1>

            <form data-formfor-todo="Todo.create" data-event-submit="todo.save">
              <input class="new-item" placeholder="add a todo item" data-bind="todo.body" />
            </form>

            <ul id="items">
                <li data-foreach-todo="Todo.all" data-mixin="animation">
                    <input type="checkbox" data-bind="todo.isDone" />
                    <label data-bind="todo.body" data-class-done="todo.isDone" data-mixin="editable"></label>
                    <a data-event-click="todo.destroy">delete</a>
                </li>
            </ul>
        </div>

        <script type="text/coffeescript">
        class Alfred extends Batman.App

        class Alfred.Todo extends Batman.Model
          @persist Batman.LocalStorage

        Alfred.run()

        if not Todo.all().length
          new Todo(body: 'joker escaped arkham again', isDone: true).save()
          new Todo(body: 'riddler sent riemann hypothesis').save()
          new Todo(body: 'bane wants to meet, not worried').save()
        </script>
    </body>
    </html>
